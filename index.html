React + Express: Fetch items from API

Express server returns items; React fetches and displays them.

// server.js (Express API)
const express = require("express");
const cors = require("cors");
const app = express();
app.use(cors());

const items = [
  { id: 1, name: "Apple" },
  { id: 2, name: "Banana" },
  { id: 3, name: "Orange" }
];

app.get("/api/items", (req, res) => {
  res.json(items);
});

app.listen(5000);


// App.js (React)
import React, { useEffect, useState } from "react";

function App() {
  const [items, setItems] = useState([]);

  useEffect(() => {
    fetch("http://localhost:5000/api/items")
      .then(response => response.json())
      .then(data => setItems(data))
      .catch(err => console.error("Error:", err));
  }, []);

  return (
    <div style={{ padding: "20px" }}>
      <h1>Items List</h1>
      <ul>
        {items.map(item => (
          <li key={item.id}>{item.name}</li>
        ))}
      </ul>
    </div>
  );
}

export default App;

useState — simple example

Small component that updates greetings as user types.

// App.js (useState example)
import React, { useState } from "react";

function App() {
  const [name, setName] = useState("");

  return (
    <div style={{ padding: "20px" }}>
      <h1>State Management Example</h1>
      <input
        type="text"
        placeholder="Enter your name"
        value={name}
        onChange={(e) => setName(e.target.value)}
      />
      <h2>Hello, {name || "Guest"}!</h2>
    </div>
  );
}

export default App;

Fetch a public API — products example

React component fetching products from Fake Store API and rendering a grid.

// App.js (React)
        

import React from "react";
import Products from "./product";

function App() {
  return (
    <div>
      <h1>Fake Store API Example</h1>
      <Products />
    </div>
  );
}

export default App;

//Product.js
  
import React, { useState, useEffect } from "react";

function Products() {
  const [products, setProducts] = useState([]);

  useEffect(() => {
    fetch("https://fakestoreapi.com/products?limit=50")
      .then((res) => res.json())
      .then((data) => setProducts(data));
  }, []);

  return (
    <div>
      <h2>Products List</h2>
      <div
        style={{
          display: "grid",
          gridTemplateColumns: "repeat(4, 1fr)",
          gap: "20px",
        }}
      >
        {products.map((product) => (
          <div
            key={product.id}
            style={{
              border: "1px solid #ccc",
              padding: "10px",
              borderRadius: "8px",
            }}
          >
            <img
              src={product.image}
              alt={product.title}
              width="100"
              height="100"
            />
            <h4>{product.title}</h4>
            <p>Price: ${product.price}</p>
            <p>Category: {product.category}</p>
            <p>
              Rating: {product.rating.rate} ⭐ ({product.rating.count} reviews)
            </p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default Products;
