Node + MongoDB: Create & Read (http server)

HTTP server that inserts into MongoDB and returns a student list.

// lab8: Node HTTP server with Mongoose (server.js)
const http = require('http');
const mongoose = require('mongoose');
const fs = require('fs');
const querystring = require('querystring');
// Connect to MongoDB
mongoose.connect('mongodb://localhost:27017/studentdb')
 .then(() => console.log('Connected to MongoDB'))
 .catch(err => console.error('MongoDB connection error:',
err));
// Define schema & model
const Student = mongoose.model('Student',
{
 name: String,
 email: String
});
// Create server
const server = http.createServer((req, res) =>
{
 // Serve the HTML form
 if (req.url === '/' && req.method === 'GET')
 {
 fs.readFile('index.html', (err, html) =>
 {
 if (err)
 {
 res.end('Error loading page');
 return;
 }
 res.writeHead(200, { 'Content-Type': 'text/html' });
 res.end(html);
 });
 }
 // Handle form submission
 else if (req.url === '/submit' && req.method === 'POST')
 {
 let body = '';
 req.on('data', chunk => body += chunk);
 req.on('end', async () =>
 {
 const data = querystring.parse(body);
 // Insert into MongoDB
 await Student.create({ name: data.name,
email: data.email });
 // Fetch all records
 const students = await Student.find();

// Display students line by line
 let result = 'Student List:\n';

students.forEach(s =>
{
 result += `${s.name} - ${s.email}\n`;
 });
 res.end(result);
 });
 }
 // 404 for other routes
 else
 {
 res.end('404 - Page Not Found');
 }
});
// Start server
server.listen(3000, () =>
{
 console.log('Server is running at
http://localhost:3000');
});

Express + MongoDB: store form data

Express app that saves posted form data to MongoDB.

// lab9: express + mongoose (server.js)
const express = require('express');
const mongoose = require('mongoose');
const bodyParser = require('body-parser');
const app = express();
const PORT = 3000;
app.use(bodyParser.urlencoded({ extended: true }));
app.use(bodyParser.json());
mongoose.connect('mongodb://localhost:27017/formdb')
.then(() => console.log('Connected to MongoDB'))
.catch(err => console.error('MongoDB connection error:',
 err));
const userSchema = new mongoose.Schema(
{
 name: String,
 email: String
});
const User = mongoose.model('User', userSchema);
app.post('/submit', (req, res) =>
{
 try
 {
 const { name, email } = req.body;
 const newUser = new User({ name, email });
 newUser.save();
 res.send(' Data saved successfully!');
 }
 catch (error)
 {
 console.error('Error saving data:', error);
 res.send('Failed to save data');
 }
});
app.listen(PORT, () => {
 console.log(` Server running at http://localhost:${PORT}`);
});

React: Simple form (useState)

React functional component capturing input and displaying submitted value.

// App.js (React)
import React, { useState } from 'react';

function UserForm() {
  const [inputValue, setInputValue] = useState('');
  const [submittedValue, setSubmittedValue] = useState('');

  const handleChange = (e) => {
    setInputValue(e.target.value);
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    setSubmittedValue(inputValue);
  };

  return (
    <div>
      <h2>Simple React Form</h2>
      <form onSubmit={handleSubmit}>
        <input
          type="text"
          placeholder="Enter something..."
          value={inputValue}
          onChange={handleChange}
        />
        <button type="submit">Submit</button>
      </form>
      {submittedValue && (
        <div>
          <strong>You entered:</strong> {submittedValue}
        </div>
      )}
    </div>
  );
}

export default UserForm;

app.use(cors());

const items = [
  { id: 1, name: "Apple" },
  { id: 2, name: "Banana" },
  { id: 3, name: "Orange" }
];

app.get("/api/items", (req, res) => {
  res.json(items);
});

app.listen(5000);


// App.js (React)
import React, { useEffect, useState } from "react";

function App() {
  const [items, setItems] = useState([]);

  useEffect(() => {
    fetch("http://localhost:5000/api/items")
      .then(response => response.json())
      .then(data => setItems(data))
      .catch(err => console.error("Error:", err));
  }, []);

  return (
    <div style={{ padding: "20px" }}>
      <h1>Items List</h1>
      <ul>
        {items.map(item => (
          <li key={item.id}>{item.name}</li>
        ))}
      </ul>
    </div>
  );
}

export default App;

// App.js (useState example)
import React, { useState } from "react";

function App() {
  const [name, setName] = useState("");

  return (
    <div style={{ padding: "20px" }}>
      <h1>State Management Example</h1>
      <input
        type="text"
        placeholder="Enter your name"
        value={name}
        onChange={(e) => setName(e.target.value)}
      />
      <h2>Hello, {name || "Guest"}!</h2>
    </div>
  );
}

export default App;

Fetch a public API — products example

React component fetching products from Fake Store API and rendering a grid.

// App.js (React)
        

import React from "react";
import Products from "./product";

function App() {
  return (
    <div>
      <h1>Fake Store API Example</h1>
      <Products />
    </div>
  );
}

export default App;

//Product.js
  
import React, { useState, useEffect } from "react";

function Products() {
  const [products, setProducts] = useState([]);

  useEffect(() => {
    fetch("https://fakestoreapi.com/products?limit=50")
      .then((res) => res.json())
      .then((data) => setProducts(data));
  }, []);

  return (
    <div>
      <h2>Products List</h2>
      <div
        style={{
          display: "grid",
          gridTemplateColumns: "repeat(4, 1fr)",
          gap: "20px",
        }}
      >
        {products.map((product) => (
          <div
            key={product.id}
            style={{
              border: "1px solid #ccc",
              padding: "10px",
              borderRadius: "8px",
            }}
          >
            <img
              src={product.image}
              alt={product.title}
              width="100"
              height="100"
            />
            <h4>{product.title}</h4>
            <p>Price: ${product.price}</p>
            <p>Category: {product.category}</p>
            <p>
              Rating: {product.rating.rate} ⭐ ({product.rating.count} reviews)
            </p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default Products;
